#include<stdio.h>
#include<string.h>
#include"global_v.h"
#include"myheaderfile.h"

void print(char output[500]){
    idx = 0;
    int is_operation = 0, count_dot = 0;
    int length_print = 0;
    rm_space_idx(output, 5);
    if(output[idx] != '(') {
        error(output, idx, "여는 괄호가 없습니다. "); //공백 무시하고 (가 없으면 에러
        return;
    }else {
        idx ++;
        rm_space_idx(output, idx);

        //문자열, 변수, 숫자 판단
        if(output[idx] == '"'){ //문자열이면

            idx ++;
            while(output[idx] != '"'){
                printf("%c", output[idx]);
                if(output[idx] == '\0' || output[idx] == ')') {
                    error(output, idx, "  \" 가 필요합니다  "); // "가 닫히지 않으면 에러
                    return;
                }else idx++;
                
            }

        }else if(output[idx] >= '0' && output[idx] <= '9' ||  //0~ 9거나
            output[idx] == '-' && output[idx+1] >= '0' && output[idx+1] <= '9' ||  // -로 시작하는 숫자거나
            output[idx] == '+' && output[idx+1] >= '0' && output[idx+1] <= '9' ){ // +로 시작하는 숫자
            for(int i = idx+1; ; i++){ //int, double / 연산인지 판단
                for(; output[i] >='0' && output[i] <='9'; i++){} //숫자가 아닌 인덱스가 나옴
                if(output[i] == '.' ) { 
                    
                    count_dot++;
                    if(count_dot >= 2) {
                        error(output, i,"잘못된 숫자 형식입니다. 점(.)은 한 번만 사용할 수 있습니다.");
                        return;
                    }
                    if(output[i+1] == ' ' || output[i+1] == ',' || output[i+1] == ')') {
                        error(output, i , "\' . \' 뒤엔 숫자가 나와야 합니다. ");
                        return;
                    }
                }
                else if (output[i-1] >= '0' && output[i-1] <= '9' && (output[i] == '+' || output[i] == '-' )|| //부호가 아닌 연산자 +, -
                output[i] == '*' || output[i] == '/' || output[i] == '%' ) { 
                    
                    if(output[rm_space(output, i+1)] < '0' || output[rm_space(output, i+1)] > '9'){
                        error(output, rm_space(output, i+1), "연산자 뒤엔 숫자가 나와야 합니다. ");
                        return;
                    }
                    is_operation ++; 
                    count_dot = 0;
                    
                }else if(output[i] == ',' || output[i] == ')') {
                    length_print = i - idx; 
                    break;
                }else if(output[i] == ' '){
                    if(get_type(output[i-1],0) == get_type(output[rm_space(output, i+1)],0)){
                        error(output, i, "이 사이엔 공백을 넣을 수 없습니다. ");
                        return;
                    }
                }
            }
            if(is_operation){
                
                double a,b;
                char print_text[500], op = '0';
                strncpy(print_text, output + idx, length_print);
                print_text[length_print] = '\0';
                int read_text = sscanf(print_text, "%lf", &a);
                for(int i = 0; i < is_operation; i++){
                    strcpy(print_text, print_text + read_text); // 1+2+3 이였으면 +2+3만 남김
                    sscanf(print_text, "%c%lf%n", &op,&b,&read_text); //op랑 b에 값을 넣음과 동시에 어디까지 읽었는지 인덱스 저장
                    switch(op){
                        case '+':
                            a += b;
                            break;
                        case '-':
                            a -= b;
                            break;
                        case '*':
                            a *= b;
                            break;
                        case '/':
                            a /= (double)b;
                            break;
                        case '%':
                            if((int)a != a || count_dot) {
                                error(output, find_char(output, idx, '%'), "실수형 자료형에서 % 연산자를 사용할 수 없습니다. ");
                                return;
                            }else a = (int)a % (int)b;
                            break;
                    }
                    printf("%g",a);

                }
            
            } else {
                while(output[idx] != ' ' && output[idx] != ',' && output[idx] != ')'){
                    printf("%c",output[idx]);
                    idx ++;
                }
            }
        }
        printf("\n");
    }
}
